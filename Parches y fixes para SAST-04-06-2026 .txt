1. Parche para src/vt_lookup.py (Sanitización de IoCs)

Modificaremos el script para que verifique que los hashes e IPs sean legítimos antes de consultar a VirusTotal:

import os
import requests
import re  # Añadimos RegEx para validación estricta

def lookup_ioc_vt(ioc_type, ioc_value):
    vt_key = os.getenv("VT_API_KEY")
    if not vt_key or "tu_api_key" in vt_key:
        return {"error": "VT_API_KEY no configurada."}

    # Limpiar espacios en blanco accidentales
    ioc_value = str(ioc_value).strip()
    ioc_type_lower = ioc_type.lower()
    endpoint = ""

    # --- DEFICIÓN DE PATRONES SEGUROS (REGEX) ---
    sha256_pattern = re.compile(r"^[a-fA-A0-9]{64}$", re.IGNORECASE)
    md5_pattern = re.compile(r"^[a-fA-A0-9]{32}$", re.IGNORECASE)
    ip_pattern = re.compile(r"^(\d{1,3}\.){3}\d{1,3}$")

    if "hash" in ioc_type_lower or "sha256" in ioc_type_lower:
        if not sha256_pattern.match(ioc_value) and not md5_pattern.match(ioc_value):
            return {"status": "error", "resultado": "Hash malformado (Bloqueado por seguridad)"}
        endpoint = f"files/{ioc_value}"
        
    elif "ipv4" in ioc_type_lower or "ip" in ioc_type_lower:
        if not ip_pattern.match(ioc_value):
            return {"status": "error", "resultado": "IP malformada (Bloqueado por seguridad)"}
        endpoint = f"ip_addresses/{ioc_value}"
        
    elif "domain" in ioc_type_lower:
        # Sanitizar caracteres peligrosos para evitar inyección de paths
        if "/" in ioc_value or ".." in ioc_value:
            return {"status": "error", "resultado": "Dominio peligroso (Bloqueado por seguridad)"}
        endpoint = f"domains/{ioc_value}"
    else:
        return None

    url = f"https://www.virustotal.com/api/v3/{endpoint}"
    headers = {"x-apikey": vt_key}

    try:
        response = requests.get(url, headers=headers, timeout=5)
        if response.status_code == 200:
            data = response.json()
            stats = data.get("data", {}).get("attributes", {}).get("last_analysis_stats", {})
            malicious = stats.get("malicious", 0)
            total = sum(stats.values())
            return {"status": "success", "resultado": f"{malicious}/{total} en VT"}
        elif response.status_code == 429:
            return {"status": "rate_limited", "resultado": "VT Rate Limit"}
        return {"status": "not_found", "resultado": "No encontrado en VT"}
    except Exception:
        return {"status": "error", "resultado": "Error de conexión VT"}
